Question 2
Email all mentors who have not assigned a final grade

Find ungraded enrollments
ungraded_enrollments = Enrollment.where(course_id: course.id, final_grade: nil)

Find mentor assignments for those enrollments
mentor_assignments = []
ungraded_enrollments.each do |enrollment|
  mentor_assignment = MentorEnrollmentAssignment.find_by(enrollment_id: enrollment.id)
  mentor_assignments << mentor_assignment if mentor_assignment
end

Get the mentors from those assignments
mentors = []
mentor_assignments.each do |assignment|
  mentor = Mentor.find(assignment.mentor_id)
  mentors << mentor if mentor
end

Find mentors id and emails
mentors.first(2).each do |mentor|
  puts "id - #{mentor.id}, email - #{mentor.email}"
end
